\begin{prob}
    次の等式を示せ。
    \begin{align*}
        \sum_{k=0}^{\lfloor n/2 \rfloor} \(\frac{n! (n-2k+1)}{(n-k+1)! k!}\)^2 = \frac{(2n)!}{n! (n+1)!}
    \end{align*}    
\end{prob}
